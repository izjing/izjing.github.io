# apply实现

# 特点

  1. 改变函数的this指向
  2. 会让函数立即执行
  3. 函数this指向apply第一个参数，第二个参数是数组的方式向函数传值

# 代码

Function.prototype.apply = function (context, args) {
    context = context ? Object(context) : window;
    context.fn = this;
    var r;
    if (!args) {
        r = context.fn();
    } esle {
        r = eval('context.fn(' + args + ')');
    }
    delete context.fn;
    return r;
};
